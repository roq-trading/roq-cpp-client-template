set(TARGET_NAME "${PROJECT_NAME}-simple")

add_subdirectory(flags)

  PRIVATE "${TARGET_NAME}-flags"
          "${PROJECT_NAME}-tools"
          roq-client::roq-client
          roq-client::roq-client-flags
          roq-logging::roq-logging
          roq-api::roq-api

          absl::flags)
